160/**
161 * @brief Sentinel FieldId representing "the current solution state" in tangent forms.
162 *
163 * When differentiating a residual form to obtain the tangent (Jacobian), undifferentiated
164 * TrialFunction occurrences are rewritten to StateField nodes. Those that represent the
165 * block's own primary unknown (rather than a named external field) use this sentinel
166 * FieldId. The assembler maps it to the current solution coefficients at each quadrature
167 * point, regardless of which physics or field variables are involved.
168 *
169 * This is distinct from INVALID_FIELD_ID, which means "uninitialized / no field."
170 * CURRENT_SOLUTION_FIELD_ID uses the same numeric value for backward compatibility
171 * with existing KernelIR encodings, but carries explicit semantic intent.

329/**
330 * @brief Strong type wrapper template for type-safe programming
331 *
332 * Prevents accidental mixing of conceptually different types that have
333 * the same underlying representation.
334 *
335 * @tparam T Underlying value type
336 * @tparam Tag Empty tag type that distinguishes otherwise identical wrappers
337 */
338template<typename T, typename Tag>
339class StrongType {
340public:
341 /** @brief Underlying value type. */
342 using ValueType = T;
343
344 /** @brief Value-initialize the wrapped value. */
345 constexpr StrongType() noexcept(std::is_nothrow_default_constructible_v<T>)
346 : value_{} {}
347
348 /**
349 * @brief Wrap an explicit value.
350 * @param value Value to store.
351 */
352 constexpr explicit StrongType(T value) noexcept(std::is_nothrow_move_constructible_v<T>)
353 : value_(std::move(value)) {}
354
355 /**
356 * @brief Access the wrapped value.
357 * @return Reference to the wrapped value.
358 */
359 constexpr T& get() noexcept { return value_; }
360 /**
361 * @brief Access the wrapped value.
362 * @return Reference to the wrapped value.
363 */
364 constexpr const T& get() const noexcept { return value_; }
365
366 /**
367 * @brief Explicitly convert back to the underlying type.
368 * @return Copy of the wrapped value.
369 */
370 constexpr explicit operator T() const noexcept { return value_; }
371
372 /**
373 * @brief Compare wrapped values for equality.
374 * @param other Wrapper to compare against.
375 * @return True when the wrapped values are equal.
376 */
377 constexpr bool operator==(const StrongType& other) const noexcept {
378 return value_ == other.value_;
379 }
380 /**
381 * @brief Compare wrapped values for inequality.
382 * @param other Wrapper to compare against.
383 * @return True when the wrapped values differ.
384 */
385 constexpr bool operator!=(const StrongType& other) const noexcept {
386 return value_ != other.value_;
387 }
388 /**
389 * @brief Order by wrapped value.
390 * @param other Wrapper to compare against.
391 * @return True when this wrapped value orders before the other.
392 */
393 constexpr bool operator<(const StrongType& other) const noexcept {
394 return value_ < other.value_;
395 }
396
397private:
398 T value_;
399};

413/**
414 * @brief DOF-specific index type
415 *
416 * @details A StrongType over GlobalIndex that prevents mixing DOF indices with
417 * other indices: conversion back to GlobalIndex is explicit (via get()), so a
418 * DofIndex cannot silently decay to a raw integer. Over the base StrongType it
419 * adds two DOF-specific conveniences -- it default-constructs to the invalid
420 * sentinel (-1) and exposes is_valid() -- for distributed DOF numbering where a
421 * negative value marks an unset or non-local DOF.
422 */
423class DofIndex : public StrongType<GlobalIndex, DofTag> {
424public:
425 using StrongType::StrongType;
426
427 /** @brief Construct an invalid DOF index (the negative sentinel). */
428 constexpr DofIndex() noexcept : StrongType(GlobalIndex{-1}) {}
429
430 /**
431 * @brief Check whether this index refers to a valid DOF.
432 * @return True when the stored value is non-negative.
433 */
434 constexpr bool is_valid() const noexcept { return get() >= 0; }
435};
