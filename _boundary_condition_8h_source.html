51/// @brief Base class for boundary conditions with spatially variable arrays
52///
53/// This class provides common functionality for boundary conditions that need to
54/// read and manage arrays of values from VTP files or uniform values. It handles
55/// distribution of data across processes and provides efficient access to values.
56///
57/// This class is intended to be subclassed by specific boundary condition types.
58///
59/// Development note: this class is intended to eventually be an object-oriented
60/// replacement of the existing bcType, although it is not yet complete.
61///
62/// Example usage:
63/// ```cpp
64/// class MyBoundaryCondition : public BoundaryCondition {
65/// public:
66/// MyBoundaryCondition(const std::string& vtp_file_path, const std::vector<std::string>& array_names, const faceType& face)
67/// : BoundaryCondition(vtp_file_path, array_names, face) {}
68///
69/// MyBoundaryCondition(const std::map<std::string, double>& uniform_values, const faceType& face)
70/// : BoundaryCondition(uniform_values, face) {}
71///
72/// // Add BC-specific functionality here
73/// };
74/// ```
75class BoundaryCondition {
76protected:
77 /// @brief Type alias for map of array names to array data
78 using StringArrayMap = std::map<std::string, Array<double>>;
79 using StringBoolMap = std::map<std::string, bool>;
80 using StringDoubleMap = std::map<std::string, double>;
81
82 /// @brief Data members for BC
83 const faceType* face_ = nullptr; ///< Face associated with the BC (not owned by BoundaryCondition)
84 int global_num_nodes_ = 0; ///< Global number of nodes on the face
85 int local_num_nodes_ = 0; ///< Local number of nodes on this processor
86 std::vector<std::string> array_names_; ///< Names of arrays to read from VTP file
87 StringArrayMap local_data_; ///< Local array values for each node on this processor
88 StringArrayMap global_data_; ///< Global array values (only populated on master)
89 StringBoolMap flags_; ///< Named boolean flags for BC behavior
90 bool spatially_variable = false; ///< Flag indicating if data is from VTP file
91 std::string vtp_file_path_; ///< Path to VTP file (empty if uniform)
92 std::map<int, int> global_node_map_; ///< Maps global node IDs to local array indices
93 std::unique_ptr<VtkVtpData> vtp_data_; ///< VTP data object
94 const SimulationLogger* logger_ = nullptr; ///< Logger for warnings/info (not owned by BoundaryCondition)
95
96public:
97 /// @brief Tolerance for point matching in VTP files
98 static constexpr double POINT_MATCH_TOLERANCE = 1e-12;
99
100 /// @brief Default constructor - creates an empty BC
101 BoundaryCondition() = default;
102
103 /// @brief Constructor - reads data from VTP file
104 /// @param vtp_file_path Path to VTP file containing arrays
105 /// @param array_names Names of arrays to read from VTP file
106 /// @param face Face associated with the BC
107 /// @param logger Simulation logger used to write warnings
108 /// @throws std::runtime_error if file cannot be read or arrays are missing
109 BoundaryCondition(const std::string& vtp_file_path, const std::vector<std::string>& array_names, const StringBoolMap& flags, const faceType& face, SimulationLogger& logger);
110
111 /// @brief Constructor for uniform values
112 /// @param uniform_values Map of array names to uniform values
113 /// @param face Face associated with the BC
114 /// @param logger Simulation logger used to write warnings
115 BoundaryCondition(const StringDoubleMap& uniform_values, const StringBoolMap& flags, const faceType& face, SimulationLogger& logger);
116
117 /// @brief Copy constructor
118 BoundaryCondition(const BoundaryCondition& other);
119
120 /// @brief Unified assignment operator (handles both copy and move)
121 BoundaryCondition& operator=(BoundaryCondition other);
122
123 /// @brief Move constructor
124 BoundaryCondition(BoundaryCondition&& other) noexcept;
125
126 /// @brief Virtual destructor
127 virtual ~BoundaryCondition() noexcept = default;
128
129 /// @brief Swap function for copy-and-swap idiom (friend function)
130 friend void swap(BoundaryCondition& lhs, BoundaryCondition& rhs) noexcept;
131
132
133 /// @brief Get value for a specific array and node
134 /// @param array_name Name of the array
135 /// @param node_id Node index on the face
136 /// @return Value for the array at the specified node
137 /// @throws std::runtime_error if array_name is not found
138 double get_value(const std::string& array_name, int node_id) const;
139
140 /// @brief Get a boolean flag by name
141 /// @param name Name of the flag
142 /// @return Value of the flag
143 /// @throws std::runtime_error if flag is not found
144 bool get_flag(const std::string& name) const;
145
146 /// @brief Get a string representation of the flags
147 /// @return String representation of the flags
148 std::string flags_to_string() const;
149
150 /// @brief Get global number of nodes
151 /// @return Global number of nodes on the face
152 int get_global_num_nodes() const noexcept {
153 return global_num_nodes_;
154 }
155
156 /// @brief Get local number of nodes
157 /// @return Local number of nodes on the face on this processor
158 int get_local_num_nodes() const noexcept {
159 return local_num_nodes_;
160 }
161
162 /// @brief Get local array index for a global node ID
163 /// @param global_node_id The global node ID defined on the face
164 /// @return Local array index for data arrays
165 /// @throws std::runtime_error if global_node_id is not found in the map
166 int get_local_index(int global_node_id) const;
167
168 /// @brief Check if data is loaded from VTP file
169 /// @return true if loaded from VTP, false if using uniform values
170 bool is_from_vtp() const noexcept {
171 return spatially_variable;
172 }
173
174 /// @brief Get the VTP file path (empty if using uniform values)
175 /// @return VTP file path
176 const std::string& get_vtp_path() const noexcept {
177 return vtp_file_path_;
178 }
179
180 /// @brief Check if this BC has been properly initialized with data
181 /// @return true if BC has data (either global or local arrays are populated)
182 bool is_initialized() const noexcept {
183 return !global_data_.empty() || !local_data_.empty();
184 }
185
186 /// @brief Distribute BC data from the master process to the slave processes
187 /// @param com_mod Reference to ComMod object for global coordinates
188 /// @param cm_mod Reference to CmMod object for MPI communication
189 /// @param cm Reference to cmType object for MPI communication
190 /// @param face Face associated with the BC
191 void distribute(const ComMod& com_mod, const CmMod& cm_mod, const cmType& cm, const faceType& face);
192
193protected:
194 /// @brief Read data from VTP file
195 /// @param vtp_file_path Path to VTP file
196 /// @param array_names Names of arrays to read
197 /// @return Map of array names to array data
198 StringArrayMap read_data_from_vtp_file(const std::string& vtp_file_path, const std::vector<std::string>& array_names);
199
200 /// @brief Find index of a point in the VTP points array
201 /// @param x X coordinate
202 /// @param y Y coordinate
203 /// @param z Z coordinate
204 /// @param vtp_points VTP points array
205 /// @param mesh_scale_factor Scale factor applied to mesh coordinates
206 /// @return Index of the matching point in the VTP array
207 /// @throws std::runtime_error if no matching point is found
208 int find_vtp_point_index(double x, double y, double z, const Array<double>& vtp_points, double mesh_scale_factor) const;
209
210 /// @brief Hook for derived classes to validate array values
211 /// @param array_name Name of the array being validated
212 /// @param value Value to validate
213 /// @throws std::runtime_error if validation fails
214 virtual void validate_array_value(const std::string& array_name, double value) const {}
215
216 // ---- distribute helpers ----
217 void distribute_metadata(const CmMod& cm_mod, const cmType& cm, bool is_slave);
218 void distribute_spatially_variable(const ComMod& com_mod, const CmMod& cm_mod, const cmType& cm, bool is_slave);
219 void distribute_uniform(const CmMod& cm_mod, const cmType& cm, bool is_slave);
220 void distribute_flags(const CmMod& cm_mod, const cmType& cm, bool is_slave);
221
222};

224/// @brief Base exception class for BC errors
225///
226/// These exceptions indicate fatal errors that should terminate the solver.
227/// They should not be caught and handled gracefully - the solver should exit.
228class BoundaryConditionBaseException : public std::exception {
229public:
230 /// @brief Constructor
231 /// @param msg Error message
232 explicit BoundaryConditionBaseException(const std::string& msg) : message_(msg) {}
233
234 /// @brief Get error message
235 /// @return Error message
236 const char* what() const noexcept override {
237 return message_.c_str();
238 }
239
240private:
241 std::string message_;
242};
