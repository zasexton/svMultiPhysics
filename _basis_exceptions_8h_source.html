23/**
24 * @brief Base exception type for errors originating in the Basis module
25 *
26 * @details The Basis module raises these basis-specific types -- rather than the
27 * generic FE exceptions in FEException.h -- so a caller can catch a basis failure
28 * precisely: an unsupported element/order pairing, a non-unisolvent node set, an
29 * out-of-range reference-node index. They all derive from FEException, so code
30 * that only wants "some FE error" can still catch the base type, and they carry
31 * the same StatusCode and source location as the rest of the hierarchy.
32 */
33class BasisException : public FEException {
34public:
35 explicit BasisException(const std::string& message,
36 StatusCode status = StatusCode::Unknown)
37 : FEException(message, status) {}
38};
39
40/**
41 * @brief Invalid Basis request or configuration
42 *
43 * @details Raised when a request is malformed before any geometry is built: a
44 * missing or negative polynomial order, a named element layout paired with an
45 * explicit order that does not match its fixed order, or a field type or
46 * continuity the scalar Lagrange/Serendipity factory does not support. Example:
47 * constructing a LagrangeBasis for Tetra10 at order 1, when that layout is fixed
48 * at order 2.
49 */
50SVMP_DEFINE_EXCEPTION(BasisConfigurationException, BasisException,
51 StatusCode::InvalidArgument);
52
53/**
54 * @brief Requested element topology is incompatible with the basis family
55 *
56 * @details Raised when the family cannot represent the requested topology or
57 * named layout. Example: requesting wedge serendipity through the arbitrary-order
58 * topology path (only the named Wedge15 layout is supported), or requesting a
59 * basis on ElementType::Unknown.
60 */
61SVMP_DEFINE_EXCEPTION(BasisElementCompatibilityException, BasisException,
62 StatusCode::InvalidArgument);
63
64/**
65 * @brief Basis evaluation request cannot be satisfied
66 *
67 * @details Raised at evaluation time rather than construction time. Example: an
68 * output span smaller than size(), or requesting analytical gradients or Hessians
69 * from a basis that does not provide them.
70 */
71SVMP_DEFINE_EXCEPTION(BasisEvaluationException, BasisException,
72 StatusCode::InvalidArgument);
73
74/**
75 * @brief Public-to-canonical node ordering or coordinate lookup failure
76 *
77 * @details Raised when a node index or coordinate lookup falls outside the
78 * reference layout. Example: requesting a tensor-axis node index outside
79 * [0, order] from line_coord_pm_one.
80 */
81SVMP_DEFINE_EXCEPTION(BasisNodeOrderingException, BasisException,
82 StatusCode::InvalidArgument);
83
84/**
85 * @brief Internal basis construction or transform setup failure
86 *
87 * @details Signals a violated internal invariant during setup (StatusCode::
88 * InternalError) rather than bad user input. Example: a generated Lagrange node
89 * lattice whose index components fall outside [0, order] in get_lagrange_lattice.
90 */
91SVMP_DEFINE_EXCEPTION(BasisConstructionException, BasisException,
92 StatusCode::InternalError);
93
94/** @} */
95
96} // namespace svmp::FE::basis
